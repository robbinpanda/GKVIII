作业简介与评估指标

本次实践作业要求同学完成一个小型图像向量检索流程。基本思想是：先选择一个图像数据集，
然后使用图像或多模态 embedding 模型将图片转换为高维向量，再使用 FAISS 等开源 ANNS 库构建
近似最近邻索引，最后对 query 图片进行检索并分析结果。
本作业关注的核心问题包括：
1. 如何将图片转换为向量表示；
2. 如何将全量向量划分为 base 向量集合与 query 向量集合；
3. 如何生成 ground truth；
4. 如何使用 FAISS 中的不同 ANNS 索引完成向量搜索；
5. 如何统计索引构建时间、索引大小、QPS 与 Recall@10；
6. 如何对检索结果进行可视化与分析。

def write_path_csv(path, paths):
"""保存向量编号与图片路径的对应关系。"""
with open(path, "w", encoding="utf-8", newline="") as f:
writer = csv.writer(f)
writer.writerow(["vector_id", "path"])
# vector_id 是该图片在对应 fvecs 文件中的行号。
for i, p in enumerate(paths):
writer.writerow([i, p])

def split_and_save(vectors, image_paths, query_size=10, seed=42):
"""将全量向量划分为 query 和 base，并保存为 fvecs。"""
# 步骤 1：确保向量数据类型为 float32，便于后续 FAISS 使用。
vectors = np.asarray(vectors, dtype="float32")
n = vectors.shape[0]
# 步骤 2：固定随机种子，保证每次实验划分结果可复现。
random.seed(seed)
# 步骤 3：随机选择 query_size 条作为 query。
query_ids = sorted(random.sample(range(n), query_size))
query_id_set = set(query_ids)
# 步骤 4：其余向量作为 base，注意 query 不得出现在 base 中。
base_ids = [i for i in range(n) if i not in query_id_set]
# 步骤 5：按照编号取出 query/base 向量与图片路径。
query_vectors = vectors[query_ids]
base_vectors = vectors[base_ids]
query_paths = [image_paths[i] for i in query_ids]
base_paths = [image_paths[i] for i in base_ids]
# 步骤 6：保存向量文件和路径映射文件。
write_fvecs("pet_query.fvecs", query_vectors)
write_fvecs("pet_base.fvecs", base_vectors)
write_path_csv("pet_query_path.csv", query_paths)
write_path_csv("pet_base_path.csv", base_paths)

import numpy as np

def generate_groundtruth(
query_fvecs_path="pet_query.fvecs",
base_fvecs_path="pet_base.fvecs",
output_ivecs_path="pet_groundtruth.ivecs",
k=10,
):
"""使用精确检索生成 ground truth top-k。"""
# 步骤 1：读取 query 向量和 base 向量。
query_vectors = read_fvecs(query_fvecs_path).astype("float32")
base_vectors = read_fvecs(base_fvecs_path).astype("float32")
# 步骤 2：检查 query 和 base 的向量维度是否一致。
nq, dq = query_vectors.shape
nb, db = base_vectors.shape
if dq != db:
raise ValueError("query and base vectors have different dimensions")
if k > nb:
raise ValueError("k cannot be larger than the number of base vectors")
# 步骤 3：使用 IndexFlatL2 建立精确检索索引。
# IndexFlatL2 会对 query 与所有 base 向量计算 L2 距离。
index = faiss.IndexFlatL2(db)
index.add(base_vectors)
# 步骤 4：对每条 query 搜索 top-k 个最近邻。
distances, indices = index.search(query_vectors, k)
# 步骤 5：将最近邻编号保存为 ivecs 文件。
# indices 中的编号是 base 向量集合中的编号。
write_ivecs(output_ivecs_path, indices.astype("int32"))
print("ground truth saved to:", output_ivecs_path)
print("query count:", nq, "base count:", nb, "topK:", k)

import os
import time
import faiss
import numpy as np

def recall_at_10(result_ids, gt_ids):
"""计算单条 query 的 Recall@10。"""
# result_ids 是 ANNS 方法返回的 top10 编号。
result_set = set(int(x) for x in result_ids[:10])
# gt_ids 是精确检索得到的 ground truth top10 编号。
gt_set = set(int(x) for x in gt_ids[:10])
# Recall@10 = 命中的 ground truth 数量 / 10。
return len(result_set & gt_set) / 10.0

def mean_recall_at_10(all_results, all_gt):
"""计算所有 query 的平均 Recall@10。"""
values = []
for i in range(len(all_results)):
values.append(recall_at_10(all_results[i], all_gt[i]))
return sum(values) / len(values)

from pathlib import Path
import numpy as np

def read_fvecs(path):
"""读取 fvecs 文件，返回 shape 为 [N, d] 的 float32 向量矩阵。"""
path = Path(path)
# 步骤 1：按 float32 读取整个二进制文件。
raw = np.fromfile(path, dtype="float32")
if raw.size == 0:
raise RuntimeError(f"empty fvecs file: {path}")
# 步骤 2：每条 fvecs 记录的第一个 4 字节是 int32 维度 d。
d = raw.view("int32")[0]
# 步骤 3：检查文件长度是否符合 [d + d 个 float32] 的重复结构。
if raw.size % (d + 1) != 0:
raise RuntimeError("invalid fvecs file format")
# 步骤 4：reshape 后去掉每行第一个维度字段，只保留真正的向量值。
raw = raw.reshape(-1, d + 1)
return raw[:, 1:].astype("float32")

def read_ivecs(path):
"""读取 ivecs 文件，返回 shape 为 [N, k] 的 int32 编号矩阵。"""
path = Path(path)
# 步骤 1：ivecs 中所有内容都是 int32。
raw = np.fromfile(path, dtype="int32")
if raw.size == 0:
raise RuntimeError(f"empty ivecs file: {path}")
# 步骤 2：每条记录的第一个 int32 是该行长度 k。
k = raw[0]
# 步骤 3：检查文件长度是否符合 [k + k 个 id] 的重复结构。
if raw.size % (k + 1) != 0:
raise RuntimeError("invalid ivecs file format")
# 步骤 4：reshape 后去掉每行第一个长度字段，只保留编号。
raw = raw.reshape(-1, k + 1)
return raw[:, 1:].astype("int32")
